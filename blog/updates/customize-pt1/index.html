www

Add an assets: section to _config.yml.

assets:
  sources:
  - _assets/vendor
  digest: true
  compression: true
  gzip: false
  defaults:

    css: { integrity: false } # true on JEKYLL_ENV=production

  compress:
    css: false
    js: uglifier
